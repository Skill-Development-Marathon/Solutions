2610. Convert an Array Into a 2D Array With Conditions


```typescript
function findMatrix(nums: number[]): number[][] {
    let sortedMatrix:number[][]=[]
    let copiedNums=[...nums]

    const arrMinifier=()=>{
        let sortedNums=[];
        for(let i=0;i<copiedNums.length;i++){
            const item=copiedNums[i]
            if(!sortedNums.includes(item)){
                sortedNums.push(item)
                copiedNums.splice(i,1)
                i=i-1
            }
        }
        sortedMatrix.push(sortedNums);
        copiedNums?.length&&arrMinifier()
    }
    arrMinifier()
    return sortedMatrix
};```
